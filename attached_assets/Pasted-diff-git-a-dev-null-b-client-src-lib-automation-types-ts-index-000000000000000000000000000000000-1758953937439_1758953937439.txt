+/**
+ * Centralizes shared automation domain types spanning personas, voice packs, posts, and
+ * humanization utilities so that they can evolve without circular imports.
+ */
+
+export type Daypart = "morning" | "midday" | "afternoon" | "evening" | "overnight";
+
+export interface DaypartPreference {
+  readonly daypart: Daypart;
+  readonly weight: number;
+}
+
+export type OffsetRange = readonly [number, number];
+
+export interface SchedulingWindow {
+  readonly daypart: Daypart;
+  readonly startHour: number;
+  readonly endHour: number;
+}
+
+export interface VoicePack {
+  readonly id: string;
+  readonly label: string;
+  readonly description: string;
+  readonly caseStyle: "original" | "sentence" | "title" | "upper" | "lower";
+  readonly prefix?: string;
+  readonly suffix?: string;
+  readonly preferredEmojis: ReadonlyArray<string>;
+  readonly disallowedPhrases: ReadonlyArray<string>;
+}
+
+export interface HumanQuirkOptions {
+  readonly emojiFrequency: "none" | "low" | "medium" | "high";
+  readonly allowTypos: boolean;
+  readonly fillerWords: ReadonlyArray<string>;
+  readonly signOff?: string;
+  readonly maxTyposPerPost?: number;
+}
+
+export interface PersonaScheduling {
+  readonly timezone: string;
+  readonly defaultWindow: SchedulingWindow;
+  readonly additionalWindows?: ReadonlyArray<SchedulingWindow>;
+  readonly daypartWeights: ReadonlyArray<DaypartPreference>;
+  readonly randomOffsetMinutes: OffsetRange;
+}
+
+export interface Persona {
+  readonly id: string;
+  readonly displayName: string;
+  readonly summary: string;
+  readonly organization?: string;
+  readonly expertise: ReadonlyArray<string>;
+  readonly voicePack: VoicePack;
+  readonly humanQuirkOptions: HumanQuirkOptions;
+  readonly scheduling: PersonaScheduling;
+  readonly complianceTags: ReadonlyArray<string>;
+}
+
+export interface Post {
+  readonly id: string;
+  readonly body: string;
+  readonly createdAt: Date;
+}
+
+export interface ScheduledPost extends Post {
+  readonly scheduledFor: Date;
+}
+
+export interface PostMetrics {
+  readonly impressions: number;
+  readonly clicks: number;
+  readonly likes: number;
+  readonly shares: number;
+  readonly comments: number;
+}
